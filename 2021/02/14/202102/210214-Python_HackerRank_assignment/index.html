이 문제는 주어진 클래스의 상속관계를 완성하고, 주어진 조건으로 내부에 method를 완성시키면 되는 간단한 문제이다.

class Student(Person):

    #   Class Constructor
    #
    #   Parameters:
    #   firstName - A string denoting the Person's first name.
    #   lastName - A string denoting the Person's last name.
    #   id - An integer denoting the Person's ID number.
    #   scores - An array of integers denoting the Person's test scores.
    #
    # Write your constructor here
    def __init__(self, _firstName, _lastName, _id, _scores):
        Person.__init__(self, _firstName, _lastName, _id)
        self.scores = _scores
    #   Function Name: calculate
    #   Return: A character denoting the grade.
    #
    # Write your function here
    def calculate(self):
        average = sum(self.scores) / len(self.scores)
        if 90 <= average <= 100:
            grade = "O"
        elif 80 <= average < 90:
            grade = "E"
        elif 70 <= average < 80:
            grade = "A"
        elif 55 <= average < 70:
            grade = "P"
        elif 40 <= average < 55:
            grade = "D"
        else:
            grade = "T"
        return grade

문제 2

Problem2 Link(HackerRank)

이 문제는 추상클래스 관련 문제로, Book 추상 클래스를 MyBook에서 구현하는 문제이다.

from abc import ABCMeta, abstractmethod
class Book:
    __metaclass__ = ABCMeta
    def __init__(self,title,author):
        self.title=title
        self.author=author
    @abstractmethod
    def display(): pass

#Write MyBook class
class MyBook(Book):
    def __init__(self, _title, _author, _price):
        Book.__init__(self, _title, _author)
        self.price = _price

    def display(self):
        print("Title: {}".format(self.title))
        print("Author: {}".format(self.author))
        print("Price: {}".format(self.price))
